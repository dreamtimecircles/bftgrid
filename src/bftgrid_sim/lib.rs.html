37#[derive(Clone, Debug)]
38pub enum SimulationEvent {
39    ClientSend {
40        // Injected before the simulation starts, is assigned an instant and reinjected as ClientRequest
41        to_node: Arc<String>,
42        event: Box<dyn ActorMsg>,
43    },
44    ClientRequest {
45        // Sent as internal event to a node's client_request_handler
46        to_node: Arc<String>,
47        event: Box<dyn ActorMsg>,
48    },
49    P2PSend {
50        // Produced by the simulated network, is assigned an instant and reinjected as P2PRequest
51        to_node: Arc<String>,
52        event: Box<dyn ActorMsg>,
53    },
54    P2PRequest {
55        // Sent as internal event to a node's p2p_request_handler
56        node: Arc<String>,
57        event: Box<dyn ActorMsg>,
58    },
59    Internal {
60        // Produced by internal modules for internal modules
61        event: InternalEvent,
62    },
63    SimulationEnd,
64}

66#[derive(Clone, Debug)]
67pub struct SimulationEventAtInstant {
68    instant: Instant,
69    event: SimulationEvent,
70}

72impl PartialEq for SimulationEventAtInstant {
73    fn eq(&self, other: &Self) -> bool {

80impl Ord for SimulationEventAtInstant {
81    fn cmp(&self, other: &Self) -> std::cmp::Ordering {

83    }
84}

177impl Simulation {
178    pub fn new(topology: Topology, start_instant: Instant, end_instant: Instant) -> Simulation {
179        Simulation {
180            topology: Arc::new(Mutex::new(topology)),
181            exited_actors: Default::default(),
182            internal_events_buffer: Default::default(),
183            events_queue: Default::default(),
184            clock: Arc::new(Mutex::new(SimulatedClock {
185                current_instant: start_instant,
186            })),
187            random: ChaCha8Rng::seed_from_u64(SEED),
188            end_instant,
189            tokio_runtime: Arc::new(
190                tokio::runtime::Builder::new_multi_thread()
191                    .enable_all()
192                    .build()
193                    .unwrap(),
194            ),
195        }
196    }
197
198    pub fn client_send<MsgT>(&mut self, to_node: String, message: MsgT)
199    where
200        MsgT: ActorMsg + 'static,
201    {
202        let instant = self.instant_of_client_request_send();
203        self.events_queue
204            .lock()
205            .unwrap()
206            .push(SimulationEventAtInstant {
207                instant,
208                event: SimulationEvent::ClientSend {
209                    to_node: Arc::new(to_node),
210                    event: Box::new(message),
211                },
212            })
213    }
214
215    pub fn run(mut self) -> Vec<SimulationEventAtInstant> {
216        let mut result = Vec::new();
217        self.events_queue
218            .lock()
219            .unwrap()
220            .push(SimulationEventAtInstant {
221                instant: self.end_instant,
222                event: SimulationEvent::SimulationEnd,
223            });
224
225        while !self.events_queue.lock().unwrap().is_empty() {
226            let mut mutex_buffer = self.internal_events_buffer.lock().unwrap();
227            let mut events_buffer = Vec::new();
228            mem::swap(&mut *mutex_buffer, &mut events_buffer);
229            drop(mutex_buffer);
230            for event in events_buffer {
231                let instant = self.instant_of_internal_event();
232                self.events_queue
233                    .lock()
234                    .unwrap()
235                    .push(SimulationEventAtInstant {
236                        instant,
237                        event: SimulationEvent::Internal { event },
238                    })
239            }
240
241            let e = self.events_queue.lock().unwrap().pop().unwrap();
242            let e_clone = e.clone();
243            result.push(e_clone);
244            self.clock.lock().unwrap().current_instant = e.instant;
245
246            if let ControlFlow::Break(_) = self.handle_event(e) {
247                break;
248            }
249        }
250        result
251    }
252
253    fn handle_event(&mut self, e: SimulationEventAtInstant) -> ControlFlow<()> {
254        match e.event {
255            SimulationEvent::ClientSend { to_node, event } => {
256                self.handle_client_send(to_node, event)
257            }
258            SimulationEvent::ClientRequest { to_node, event } => {
259                self.handle_client_request(to_node, event)
260            }
261            SimulationEvent::P2PSend { to_node, event } => self.handle_p2p_send(to_node, event),
262            SimulationEvent::P2PRequest { node, event } => self.handle_p2p_request(node, event),
263            SimulationEvent::Internal {
264                event:
265                    InternalEvent {
266                        node,
267                        handler,
268                        event,
269                        delay,
270                    },
271            } => self.handle_internal_event(delay, e.instant, node, handler, event),
272            SimulationEvent::SimulationEnd => return ControlFlow::Break(()),
273        }
274        ControlFlow::Continue(())
275    }
276
277    fn handle_internal_event(
278        &mut self,
279        delay: Option<Duration>,
280        instant: Instant,
281        node: Arc<String>,
282        handler: Arc<String>,
283        event: Box<dyn ActorMsg>,
284    ) {
285        match delay {
286            Some(duration) => {
287                self.events_queue
288                    .lock()
289                    .unwrap()
290                    .push(SimulationEventAtInstant {
291                        instant: instant.add(duration),
292                        event: SimulationEvent::Internal {
293                            event: InternalEvent {
294                                node,
295                                handler,
296                                event,
297                                delay: None,
298                            },
299                        },
300                    });
301            }
302            None => {
303                if self
304                    .exited_actors
305                    .lock()
306                    .unwrap()
307                    .iter()
308                    .any(|elem| *elem == handler)
309                {
310                    panic!("message sent to actor that has exited")
311                }
312                let mut topology = self.topology.lock().unwrap();
313                let mut removed_node = topology
314                    .remove(node.as_ref())
315                    .unwrap_or_else(|| panic!("node {:?} unknown", node));
316                let removed_handler_arc = removed_node
317                    .all_handlers
318                    .remove(&handler)
319                    .unwrap_or_else(|| {
320                        panic!("handler {:?} not found for node {:?}", handler, node)
321                    });
322                topology.insert((*node).clone(), removed_node);
323                drop(topology); // So that a handler with access to the actor system can lock it again in `crate` and/or `set_handler`
324                if let Some(control) = removed_handler_arc
325                    .lock()
326                    .unwrap()
327                    .as_mut()
328                    .unwrap_or_else(|| {
329                        panic!("handler {:?} not found for node {:?}", handler, node)
330                    })
331                    .receive_untyped(event.clone())
332                    .unwrap_or_else(|_| {
333                        panic!(
334                            "handler {:?} for node {:?} cannot handle event {:?}",
335                            handler, node, event
336                        )
337                    })
338                {
339                    match control {
340                        ActorControl::Exit() => self.exited_actors.lock().unwrap().push(handler),
341                    }
342                } else {
343                    let mut topology = self.topology.lock().unwrap();
344                    let mut removed_node = topology
345                        .remove(node.as_ref())
346                        .unwrap_or_else(|| panic!("node {:?} not found", node));
347                    removed_node
348                        .all_handlers
349                        .insert(handler, removed_handler_arc.clone());
350                    topology.insert((*node).clone(), removed_node);
351                };
352            }
353        }
354    }

356    fn handle_p2p_request(&mut self, node: Arc<String>, event: Box<dyn ActorMsg>) {
357        let internal_event_instant = self.instant_of_internal_event();
358        let p2p_request_handler = self
359            .topology
360            .lock()
361            .unwrap()
362            .get(node.as_ref())
363            .unwrap()
364            .p2p_request_handler
365            .as_ref()
366            .unwrap_or_else(|| panic!("p2p request handler unset for node {:?}", node))
367            .clone();
368        self.events_queue
369            .lock()
370            .unwrap()
371            .push(SimulationEventAtInstant {
372                instant: internal_event_instant,
373                event: SimulationEvent::Internal {
374                    event: InternalEvent {
375                        node,
376                        handler: p2p_request_handler,
377                        event,
378                        delay: None,
379                    },
380                },
381            })
382    }

413            .push(SimulationEventAtInstant {
414                instant: internal_event_instant,
415                event: SimulationEvent::Internal {
416                    event: InternalEvent {
417                        node: to_node,
418                        handler: client_request_handler,
419                        event,
420                        delay: None,
421                    },
422                },
423            })
424    }

426    fn handle_client_send(&mut self, to_node: Arc<String>, event: Box<dyn ActorMsg>) {
427        let client_request_arrival_instant = self.instant_of_client_request_arrival();
428        self.events_queue
429            .lock()
430            .unwrap()
431            .push(SimulationEventAtInstant {
432                instant: client_request_arrival_instant,
433                event: SimulationEvent::ClientRequest { to_node, event },
434            })

437    fn instant_of_internal_event(&mut self) -> Instant {
438        let pseudo_random_between_zero_and_one = self.random.next_u64() as f64 / u64::MAX as f64;
439        let pseudo_random_duration =
440            MAX_RANDOM_DURATION.mul_f64(pseudo_random_between_zero_and_one);
441        self.clock
442            .lock()
443            .unwrap()
444            .current_instant
445            .add(pseudo_random_duration)
446    }
447
448    fn instant_of_client_request_send(&mut self) -> Instant {
449        self.instant_of_internal_event()
450    }
451
452    fn instant_of_client_request_arrival(&mut self) -> Instant {
453        self.instant_of_internal_event()
454    }

457        self.instant_of_internal_event()
458    }
459
460    fn instant_of_p2p_request_arrival(&mut self) -> Instant {
461        self.instant_of_internal_event()
462    }

490impl ActorSystem for Simulation {
491    type ActorRefT<MsgT: ActorMsg, HandlerT: TypedHandler<MsgT = MsgT> + 'static> =
492        SimulatedActor<MsgT, HandlerT>;
493
494    fn create<MsgT, HandlerT>(
495        &mut self,
496        node_id: impl Into<String>,
497        name: impl Into<String>,
498    ) -> SimulatedActor<MsgT, HandlerT>
499    where
500        MsgT: ActorMsg,
501        HandlerT: TypedHandler<MsgT = MsgT> + 'static,
502    {
503        let node_id_string = node_id.into();
504        let name_string = name.into();
505        let handler_rc = Arc::new(Mutex::new(None));
506        let name_arc = Arc::new(name_string);
507        let name_arc2 = name_arc.clone();
508        if self
509            .topology
510            .lock()
511            .unwrap()
512            .get_mut(&node_id_string)
513            .unwrap_or_else(|| panic!("Node {:?} unknown", &node_id_string))
514            .all_handlers
515            .insert(name_arc, handler_rc.clone())
516            .is_some()
517        {
518            panic!("An actor with such a name already exist");
519        }
520        SimulatedActor {
521            topology: self.topology.clone(),
522            node_id: Arc::new(node_id_string),
523            name: name_arc2,
524            handler: handler_rc,
525            events_buffer: self.internal_events_buffer.clone(),
526            message_type: PhantomData {},
527            handler_type: PhantomData {},
528        }
529    }
530
531    fn set_handler<MsgT, HandlerT>(
532        &mut self,
533        actor_ref: &mut Self::ActorRefT<MsgT, HandlerT>,
534        handler: HandlerT,
535    ) where
536        MsgT: ActorMsg,
537        HandlerT: TypedHandler<MsgT = MsgT> + 'static,
538    {
539        let mut topology = self.topology.lock().unwrap();
540        let node = topology.get_mut(actor_ref.node_id.as_ref()).unwrap();
541        node.all_handlers.insert(
542            actor_ref.name.clone(),
543            Arc::new(Mutex::new(Some(Box::new(handler)))),
544        );
545    }
546
547    fn spawn_async_send<MsgT, HandlerT, O>(
548        &self,
549        f: impl std::prelude::rust_2024::Future<Output = O> + Send + 'static,
550        to_msg: impl FnOnce(O) -> MsgT + Send + 'static,
551        mut actor_ref: bftgrid_core::AnActorRef<MsgT, HandlerT>,
552        delay: Option<Duration>,
553    ) where
554        MsgT: ActorMsg + 'static,
555        HandlerT: TypedHandler<MsgT = MsgT> + 'static,
556    {
557        let res = self.tokio_runtime.block_on(f);
558        actor_ref.send(to_msg(res), delay);
559    }
560
561    fn spawn_blocking_send<MsgT, HandlerT, R>(
562        &self,
563        f: impl FnOnce() -> R + Send + 'static,
564        to_msg: impl FnOnce(R) -> MsgT + Send + 'static,
565        mut actor_ref: bftgrid_core::AnActorRef<MsgT, HandlerT>,
566        delay: Option<Duration>,
567    ) where
568        MsgT: ActorMsg + 'static,
569        HandlerT: TypedHandler<MsgT = MsgT> + 'static,
570        R: Send + 'static,
571    {
572        actor_ref.send(to_msg(f()), delay);
573    }
574}

576impl P2PNetworkClient for Simulation {
577    fn attempt_send<MsgT, SerializerT>(
578        &mut self,
579        message: MsgT,
580        _serializer: &SerializerT,
581        to_node: impl Into<String>,
582    ) -> P2PNetworkResult<()>
583    where
584        MsgT: ActorMsg,
585        SerializerT: Fn(MsgT) -> P2PNetworkResult<Vec<u8>> + Sync,
586    {
587        let instant = self.instant_of_p2p_request_send();
588        self.events_queue
589            .lock()
590            .unwrap()
591            .push(SimulationEventAtInstant {
592                instant,
593                event: SimulationEvent::P2PSend {
594                    to_node: Arc::new(to_node.into()),
595                    event: Box::new(message),
596                },
597            });
598        Ok(())
599    }
