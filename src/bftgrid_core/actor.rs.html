1//! An actor framework geared towards BFT ordering libraries with support for deterministic simulation testing.
2//! It also allows writing a generic overall system construction logic independent from the specific actor system used.
3//! Every actor is managed by a single actor system but can interact with actors managed by other actor systems.
4//!
5//! No async signatures are used in the public API in order to support single-threaded simulation testing without relying on async runtimes.
6//! Actors should not assume anything about the thread they are running on, nor use any async runtime. They must rely on actor system facilities
7//! to execute thread-blocking and async tasks.
8

22pub trait ActorMsg: DynClone + Any + Send + Debug {}
23clone_trait_object!(ActorMsg);

25pub enum ActorControl {
26    Exit(),
27}

29/// A [`TypedMsgHandler`] is an actor behavior that can handle messages of a specific type
30/// and optionally return an [`ActorControl`] message.
31pub trait TypedMsgHandler<MsgT>: Send + Debug
32where
33    MsgT: ActorMsg,
34{
35    fn receive(&mut self, message: MsgT) -> Option<ActorControl>;
36}
37
38pub type MsgHandler<MsgT> = Box<dyn TypedMsgHandler<MsgT>>;
39
40#[derive(Debug, Clone)]
41pub struct MessageNotSupported();
42
43impl Display for MessageNotSupported {
44    fn fmt(&self, f: &mut Formatter) -> Result<(), FmtError> {
45        write!(f, "Message not supported")
46    }
47}
48
49impl Error for MessageNotSupported {}
50
51pub type AnActorMsg = Box<dyn ActorMsg>;
52
53/// An [`UntypedMsgHandler`] is an actor handler that can receive messages of any type,
54///  although it may refuse to handle some of them.
55pub trait UntypedMsgHandler: Send + Debug {
56    fn receive_untyped(
57        &mut self,
58        message: AnActorMsg,
59    ) -> Result<Option<ActorControl>, MessageNotSupported>;
60}
61
62pub type UntypedHandlerBox = Box<dyn UntypedMsgHandler>;
63
64/// A blanket [`UntypedMsgHandler`] implementation for any [`MsgHandler<MsgT>`]
65///  to allow any boxed typed actor to be used as a network input actor.
66// The manual supertrait upcasting approach
67//  (https://quinedot.github.io/rust-learning/dyn-trait-combining.html#manual-supertrait-upcasting)
68//  would not work in this case due to the `MsgT` type parameter being
69//  method-bound rather than trait-bound in a blanket implementation for `T: TypedMsgHandler<MsgT>`.
70//  Somewhat sadly, this means that a generic [`UntypedMsgHandler`] must be costructed via a second level
71//  of boxing (`Box<dyn UntypedMsgHandler>`).
72impl<MsgT> UntypedMsgHandler for MsgHandler<MsgT>
73where
74    MsgT: ActorMsg,
75{
76    fn receive_untyped(
77        &mut self,
78        message: AnActorMsg,
79    ) -> Result<Option<ActorControl>, MessageNotSupported> {
80        match (message as Box<dyn Any>).downcast::<MsgT>() {
81            Ok(typed_message) => Result::Ok(self.receive(*typed_message)),
82            Err(_) => Result::Err(MessageNotSupported()),
83        }
84    }
85}

87/// A [`Task`] can be queried for completion.
88pub trait Task: Send + Debug {
89    fn is_finished(&self) -> bool;
90}

92/// A [`Joinable`] can be awaited for completion in a thread-blocking fashion.
93/// Specific types of [`ActorRef`] and [`ActorSystemHandle`] can be joined to wait for their completion.
94pub trait Joinable<Output>: Task + Send + Debug {
95    fn join(&mut self) -> Output;
96}

98/// An [`ActorRef`] can asynchronously send messages to the underlying actor, optionally with a delay, and create new actor references.
99/// Actor implementations can use [`ActorRef`]s to send messages to themselves and to other actors.
100pub trait ActorRef<MsgT>: Clone + Send + Debug
101where
102    MsgT: ActorMsg + 'static,
103{
104    fn send(&mut self, message: MsgT, delay: Option<Duration>);
105}
106
107/// An [`ActorSystemHandle`] allows spawning actors by creating an [`ActorRef`] and setting its handler.
108/// The handler of the underlying actor can also be changed at any time.
109/// An [`ActorSystemHandle`] can be cloned.
110/// Actors themselves can use [`ActorSystemHandle`]s to spawn new actors and even to change their own handlers.
111pub trait ActorSystemHandle: Clone {
112    type ActorRefT<MsgT>: ActorRef<MsgT>
113    where
114        MsgT: ActorMsg;
115
116    fn create<MsgT>(
117        &self,
118        node_id: impl Into<String>,
119        name: impl Into<String>,
120        join_on_drop: bool,
121    ) -> Self::ActorRefT<MsgT>
122    where
123        MsgT: ActorMsg;
124
125    fn set_handler<MsgT>(&self, actor_ref: &mut Self::ActorRefT<MsgT>, handler: MsgHandler<MsgT>)
126    where
127        MsgT: ActorMsg;
128
129    fn spawn_async_send<MsgT>(
130        &self,
131        f: impl Future<Output = MsgT> + Send + 'static,
132        actor_ref: impl ActorRef<MsgT> + 'static,
133        delay: Option<Duration>,
134    ) where
135        MsgT: ActorMsg + 'static;
136
137    fn spawn_thread_blocking_send<MsgT>(
138        &self,
139        f: impl FnOnce() -> MsgT + Send + 'static,
140        actor_ref: impl ActorRef<MsgT> + 'static,
141        delay: Option<Duration>,
142    ) where
143        MsgT: ActorMsg + 'static;
144}

146#[derive(Error, Debug, Clone)]
147pub enum P2PNetworkError {
148    #[error("I/O error")]
149    Io(#[from] Arc<io::Error>),
150    #[error("Join error")]
151    Join(#[from] Arc<JoinError>),
152    #[error("Actor not found")]
153    ActorNotFound(Arc<String>),
154}
155
156pub type P2PNetworkResult<R> = Result<R, P2PNetworkError>;
157
158/// A [`P2PNetworkClient`] allows sending messages to other nodes in a P2P network.
159pub trait P2PNetworkClient: Clone {
160    fn attempt_send<MsgT, SerializerT>(
161        &mut self,
162        message: MsgT,
163        serializer: &SerializerT,
164        node: impl Into<String>,
165    ) -> P2PNetworkResult<()>
166    where
167        MsgT: ActorMsg,
168        SerializerT: Fn(MsgT) -> P2PNetworkResult<Vec<u8>> + Sync;
169}
