24fn notify_close(close_cond: Arc<(Mutex<bool>, Condvar)>) {
25    let (closed_mutex, cvar) = &*close_cond;
26    let mut closed = closed_mutex.lock().unwrap();
27    *closed = true;
28    cvar.notify_all();
29}
